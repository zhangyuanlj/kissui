JS代码
//加载列表，需要导入page代码包
var pageSize = 2;
var page = new KPage({
    insertObj : "#pageHtml",
    openInfor : false,  
    pageCount : 6,
    pageSize : pageSize,
     callback : function(){
        loadList();
    }
});
page.init();
loadList();
//加载列表
function loadList(){
    var resultHtml = [];
    var pageNum = page.getPageNum();
    $.ajax({
        type : "GET",
        url : "news.php",
        data:"page=" + pageNum + "&pageSize=" + pageSize,
        cache:false,
        beforeSend:function(){
            $("#loading").show();
        },
        success: function(xml){
            $("#loading").hide()
            var listTag = "items";
            var dataNum = parseInt($(xml).find("dataNum").text());
            var dataLength = $(xml).find(listTag).length;
            for(var i = 0; i < dataLength; i++){
                var title = $(xml).find(listTag + ":eq(" + i +") title").text();
                resultHtml.push("<p>"+title+"</p>");
            }
            $("#news_list").html(resultHtml.join("\n"));
            page.dataNum = dataNum;
            page.updatePage();
        },
        error:function(){
            $("#loading").hide()
            alert("加载数据失败,请稍后...");
        }
    });
}
配置说明
属性 说明 是否必填
insertObj 插入分页的目标对象 必填
pageSize 每页显示多少条记录 可选
pageCount 每段显示的页码 可选
offsetPage 页码偏移数 可选
prevText 上一页链接文字 可选
nextText 下一页链接文字 可选
openInfor 是否开启详细信息 可选
callback 前往目标页码执行的回调函数 必填
KPage方法
方法名 作用
init 初始化
getPageNum 获取当前页码
goPrev 上一页
goNext 下一页
goTargetPage(integer pageNum) 跳转到指定的页码
updatePage 更新分页控件
